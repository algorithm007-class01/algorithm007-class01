class Solution {
    public int[] twoSum(int[] nums, int target) {
        int[] resultArr=new int[2];
        for(int i=0;i<nums.length-1;i++){
            for(int j=i+1;j<nums.length;j++){
                int result=nums[i]+nums[j];
                if(result==target){
                  resultArr[0]=i;
                  resultArr[1]=j;
                }
            }
        }
        return resultArr;
    }
}
